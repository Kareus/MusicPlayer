421 int AVLTree<T>::DeleteNode(AVLTreeNode<T>*& node, const T& data)
422 {
423  if (node == nullptr) return 0;
424  int success = 0;
425 
426  int compare = compareFunc(node->data, data);
427 
428  if (compare > 0) //data가 작은 경우
429  success = DeleteNode(node->left, data);
430  else if (compare < 0) //data가 큰 경우
431  success = DeleteNode(node->right, data);
432  else //찾은 경우
433  {
434  success = 1;
435  if (node->left && node->right) //child가 둘인 경우
436  {
437  AVLTreeNode<T>* r = node->right;
438  AVLTreeNode<T>* n = node; //r의 부모 노드
439 
440  bool leftChild = false;
441 
442  while (r->left)
443  {
444  n = r;
445  r = r->left;
446  leftChild = true;
447  }
448  //현재 data보다 큰 값 중 가장 작은 값을 찾는다. (되도록 complete tree를 만들기 위함)
449  //r은 leaf node임이 보장된다.
450 
451  node->data = r->data;
452  delete r;
453 
454  if (leftChild) n->left = nullptr; //부모 노드가 dangling pointer를 가리키지 않도록 방지
455  else n->right = nullptr; //right node가 leaf node인 경우
456  }
457  else //child가 없거나 하나인 경우
458  {
459  AVLTreeNode<T>* temp = node;
460  if (node->left == nullptr) node = node->right;
461  else if (node->right == nullptr) node = node->left;
462  delete temp;
463 
464  if (root == temp) root = nullptr; //최상위 노드인 경우 root가 dangling pointer를 가리키지 않도록 방지 (노드가 root 하나 뿐이므로)
465  //이외 발생하는 dangling pointer는 회전 과정에서 해결된다.
466 
467  return 1; //child가 하나인 경우엔 node가 overwrite되고, 이 node는 leaf node가 되므로 회전 과정 불필요
468  //child가 없는 경우엔 node가 delete되고 nullptr이 됨. 회전 과정이 불필요할 뿐더러, 접근 자체가 불가능
469  }
470  }

472  CalculateHeight(node); //삭제 실행 후 height을 계산
473 
474  if (height(node->left) - height(node->right) == 2) //왼쪽 불균형
475  {
476  if (height(node->left->left) - height(node->left->right) == 1) //왼쪽의 왼쪽 (left left case)
477  node = SingleRightRotate(node);
478  else //왼쪽의 오른쪽 (left right case)
479  node = DoubleRightRotate(node);
480  }
481  else if (height(node->right) - height(node->left) == 2) //오른쪽 불균형
482  {

484  node = SingleLeftRotate(node);
485  else //오른쪽의 왼쪽 (right left case)
486  node = DoubleLeftRotate(node);
